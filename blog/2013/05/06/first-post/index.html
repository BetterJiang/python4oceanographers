dk = np.arange(0.01, 2.5, 0.01)

C1, C2, C3 = [], [], []

for i in np.arange(np.size(dk)):
    C1.append(K[0].subs(k, dk[i]))
    C2.append(K[1].subs(k, dk[i]))
    C3.append(K[2].subs(k, dk[i]))
    
C1, C2, C3 = map(lambda x: np.asanyarray(x, np.complex_), (C1, C2, C3))
