Namespaces

import this

Direto da wikipedia:

A namespace (sometimes also called a name scope) is an abstract container or environment created to hold a logical grouping of unique identifiers or symbols (i.e., names).

O que interessa para nós?

  • Namespaces mantém o código organizado.
  • Evita conflitos gerados por funções de mesmo nome.
  • Torna o código "declarativo" e consequentemente mais legível.

Como rodar um script?

Rodar programa completo.

run plt_map.py

  1 # -*- coding: utf-8 -*-
  2 #
  3 # plt_map.py
  4 #
  5 # purpose:  Read all stations and plot a map
  6 # author:   Filipe P. A. Fernandes
  7 # e-mail:   ocefpaf@gmail
  8 # web:      http://ocefpaf.tiddlyspot.com/
  9 # created:  13-Nov-2012
 10 # modified: Wed 14 Nov 2012 04:03:00 PM BRST
 11 #
 12 # obs: Plot a map and a depth profile for the CTD transect.
 13 #
 14 
 15 import os
 16 from glob import glob
 17 
 18 import numpy as np
 19 import seawater.csiro as sw
 20 import matplotlib.pyplot as plt
 21 from netCDF4 import Dataset
 22 from mpl_toolkits.basemap import Basemap, shiftgrid, cm
 23 
 24 def break_lines(line):
 25     return [float(num) for num in line.strip().split()]
 26 
 27 def get_topo(url, m):
 28     etopodata = Dataset(url)
 29     topoin = etopodata.variables['ROSE'][:]
 30     lons = etopodata.variables['ETOPO05_X'][:]
 31     lats = etopodata.variables['ETOPO05_Y'][:]
 32     topoin, lons = shiftgrid(180., topoin, lons, start=False)
 33     nx = int((m.xmax - m.xmin) / 5000.) + 1
 34     ny = int((m.ymax - m.ymin) / 5000.) + 1
 35     return m.transform_scalar(topoin, lons, lats, nx, ny)
 36 
 37 # Shelf angle.
 38 def bathymetry_angle(start=0, end=9):
 39     opposite = depth[end] - depth[start]
 40     adjacent = dist[end] - dist[start]
 41     return np.arctan2(-opposite, adjacent) * 180 / np.pi
 42 
 43 lista = glob(os.path.join('..', 'data', '*.dat'))
 44 lista.sort()
 45 
 46 depth, lat, lon = [], [], []
 47 for fname in lista:
 48     with open(fname, 'r') as f:
 49         lines = f.readlines()
 50         d, la, lo = break_lines(lines[0])[3:]
 51         lon.append(lo)
 52         lat.append(la)
 53         depth.append(d)
 54 
 55 tangent = bathymetry_angle(0, 9)
 56 print("Shelf angle: %s" % tangent)
 57 
 58 # Map.
 59 offset = 2.
 60 llcrnrlon, urcrnrlon = min(lon) - offset, max(lon) + offset
 61 llcrnrlat, urcrnrlat = min(lat) - offset, max(lat) + offset
 62 
 63 m = Basemap(projection='merc',
 64             llcrnrlon=llcrnrlon,
 65             urcrnrlon=urcrnrlon,
 66             llcrnrlat=llcrnrlat,
 67             urcrnrlat=urcrnrlat,
 68             lat_ts=20, resolution='h')
 69 fig, ax = plt.subplots(figsize=(20, 20), facecolor='w')
 70 m.ax = ax
 71 
 72 url = 'http://ferret.pmel.noaa.gov/thredds/dodsC/data/PMEL/etopo5.nc'
 73 topodat = get_topo(url, m)
 74 
 75 im = m.imshow(topodat, cm.GMT_haxby)
 76 m.drawstates()
 77 m.drawcountries()
 78 m.drawcoastlines()
 79 m.plot(lon, lat, 'r*', latlon=True)
 80 cb = m.colorbar(im, 'right', size='5%', pad='2%')
 81 meridians = np.arange(int(llcrnrlon), int(urcrnrlon), 1.)
 82 parallels = np.arange(int(llcrnrlat), int(urcrnrlat), 1.)
 83 kw = dict(fontsize=20, fontweight='demibold')
 84 m.drawparallels(parallels, labels=[1, 0, 0, 1], **kw)
 85 m.drawmeridians(meridians, labels=[1, 1, 0, 1], **kw)
 86 
 87 # Profile.
 88 dist, phaseangle = sw.dist(lon, lat, units='km')
 89 dist = np.r_[0, np.cumsum(dist)] * 1e3
 90 
 91 fig, ax = plt.subplots(figsize=(12, 9), facecolor='w')
 92 ax.fill_between(dist, max(depth), depth, color='0.75')
 93 ax.plot(dist, depth, color='k')
 94 ax.set_ylim(0, max(depth))
 95 ax.set_xlim(-5, max(dist) + 10)
 96 ax.invert_yaxis()
 97 ax.set_ylabel("Depth [db]")
 98 ax.set_xlabel("Distance [m]")
 99 ax.plot(dist, len(dist) * [0], 'kv')
100 ax.xaxis.tick_bottom()
101 
102 plt.show()

run plt_TS.py

 1 # -*- coding: utf-8 -*-
 2 #
 3 # plt_TS.py
 4 #
 5 # purpose:  Plot a TS diagram
 6 # author:   Filipe P. A. Fernandes
 7 # e-mail:   ocefpaf@gmail
 8 # web:      http://ocefpaf.tiddlyspot.com/
 9 # created:  13-Nov-2012
10 # modified: Tue 13 Nov 2012 01:11:33 PM BRST
11 #
12 # obs:
13 #
14 
15 import os
16 from glob import glob
17 
18 import numpy as np
19 import seawater.csiro as sw
20 import matplotlib.pyplot as plt
21 
22 from pandas import DataFrame
23 
24 def basename(fname):
25     return os.path.splitext(os.path.basename(fname))
26 
27 def read_station(fname):
28     P, T, S = np.loadtxt(fname, unpack=True, usecols=(0, 3, 4),
29                          skiprows=1)
30     column = basename(fname)[0]
31     return (DataFrame(T, index=P, columns=[column]),
32             DataFrame(S, index=P, columns=[column]))
33 
34 lista = glob(os.path.join('..', 'data', '*.dat'))
35 lista.sort()
36 
37 first = lista.pop(0)
38 tmp, sal = read_station(first)
39 
40 for fname in lista:
41     T, S = read_station(fname)
42     tmp = tmp.join(T, how='right')
43     sal = sal.join(S, how='right')
44 
45 Te = np.arange(0, 32, 2)
46 Se = np.arange(32, 38.25, 0.25)
47 
48 Sg, Tg = np.meshgrid(Se, Te)
49 cnt = np.arange(20, 34)
50 
51 sigma_theta = sw.pden(Sg, Tg, 0, 0) - 1000
52 
53 deg = u"\u00b0"
54 
55 fig, ax = plt.subplots(figsize=(6, 6))
56 ax.plot(sal, tmp, 'k.')
57 ax.set_ylabel("Temperatura %sC" % deg)
58 ax.set_xlabel("Salinidade [g kg$^{-1}$]")
59 
60 cs = ax.contour(Se, Te, sigma_theta, colors='black', levels=cnt)
61 ax.clabel(cs, fontsize=9, inline=1, fmt='%2.1f')
62 ax.axis([31.8, 37.2, 0.0, 30.0])
63 
64 fig.savefig("TS_diagram.svg", transparent=True)
65 
66 plt.show()
